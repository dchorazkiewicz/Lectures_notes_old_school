% Local visual language for these notes. Formal mathematical statements remain
% easy to locate, while explanatory material stays inside the reading flow.
\definecolor{CoreBlue}{HTML}{0B4F6C}
\definecolor{CoreBlueBack}{HTML}{EAF4F8}

% Definitions and theorems are the only routinely highlighted environments.
% A quiet background and a single left rule replace full coloured frames.
\renewtcolorbox{definitionbox}[1][Definition]{
    enhanced,
    breakable,
    colback=CoreBlueBack!55,
    colframe=CoreBlue,
    boxrule=0pt,
    leftrule=1.2pt,
    arc=0pt,
    outer arc=0pt,
    boxsep=0pt,
    left=8pt,
    right=6pt,
    top=5pt,
    bottom=5pt,
    before skip=0.8em,
    after skip=0.8em,
    title={#1},
    coltitle=CoreBlue,
    fonttitle=\bfseries,
    attach title to upper={\par\smallskip},
    before upper={\let\boxed\plainformula}
}

% Explanations, examples, remarks, interpretations, derivations, and summaries
% use run-in labels instead of panels. This preserves their semantic role
% without interrupting the page with repeated boxes and colour changes.
\renewenvironment{corebox}[1][Core idea]
  {\let\boxed\plainformula\par\medskip\noindent\textbf{#1.}\enspace\ignorespaces}
  {\par\medskip}

\renewenvironment{derivationbox}[1][Derivation]
  {\let\boxed\plainformula\par\medskip\noindent\textbf{#1.}\enspace\ignorespaces}
  {\par\medskip}

\renewenvironment{proofbox}[1][Proof]
  {\let\boxed\plainformula\begin{proof}[#1]}
  {\end{proof}}

\renewenvironment{interpretationbox}[1][Interpretation]
  {\let\boxed\plainformula\par\smallskip\noindent\textit{#1.}\enspace\ignorespaces}
  {\par\smallskip}

\renewenvironment{examplebox}[1][Example]
  {\let\boxed\plainformula\par\medskip\noindent\textbf{#1.}\enspace\ignorespaces}
  {\par\medskip}

\renewenvironment{remarkbox}[1][Remark]
  {\let\boxed\plainformula\par\smallskip\noindent\textit{#1.}\enspace\ignorespaces}
  {\par\smallskip}

\renewenvironment{summarybox}[1][Summary]
  {\let\boxed\plainformula\par\medskip\noindent\textbf{#1.}\enspace\ignorespaces}
  {\par\medskip}
