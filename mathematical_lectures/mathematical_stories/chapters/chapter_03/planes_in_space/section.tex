\section{Planes in Three-Dimensional Space}

A line was produced by allowing one real parameter:
\[
P=P_0+t\mathbf v.
\]
Only one independent direction of motion was available.

\begin{corebox}[The geometric question]
What set is obtained when motion from one fixed point is allowed in two
independent directions?
\end{corebox}

Let $P_0\in\mathbb R^3$ be fixed and let $\mathbf u,\mathbf v$ be nonparallel
vectors.

\begin{derivationbox}[Generating the plane]
Every permitted displacement is a linear combination
\[
s\mathbf u+t\mathbf v.
\]
Starting at $P_0$ gives
\[
P=P_0+s\mathbf u+t\mathbf v.
\]
One parameter produces a line; two independent parameters produce a plane. If
the two directions were parallel, both parameters would still move only along
one line.
\end{derivationbox}

\begin{definitionbox}
The plane through $P_0$ generated by nonparallel vectors $\mathbf u$ and
$\mathbf v$ is
\[
\boxed{
\Pi=\{P_0+s\mathbf u+t\mathbf v:s,t\in\mathbb R\}
}.
\]
If
\[
P_0=(x_0,y_0,z_0),\quad
\mathbf u=(u_1,u_2,u_3),\quad
\mathbf v=(v_1,v_2,v_3),
\]
then
\[
\boxed{
\begin{cases}
x=x_0+s u_1+t v_1,\\
y=y_0+s u_2+t v_2,\\
z=z_0+s u_3+t v_3.
\end{cases}
}
\]
\end{definitionbox}

\subsection{A Normal Vector from Two Directions}

The plane is generated by two directions. To write one equation for it, we need
a vector perpendicular to both of them.

\begin{definitionbox}
For
\[
\mathbf u=(u_1,u_2,u_3),
\qquad
\mathbf v=(v_1,v_2,v_3),
\]
their \emph{cross product} is
\[
\boxed{
\mathbf u\times\mathbf v
=
\bigl(
 u_2v_3-u_3v_2,
\;u_3v_1-u_1v_3,
\;u_1v_2-u_2v_1
\bigr)
}.
\]
\end{definitionbox}

Direct substitution shows that
\[
(\mathbf u\times\mathbf v)\cdot\mathbf u=0,
\qquad
(\mathbf u\times\mathbf v)\cdot\mathbf v=0.
\]
Therefore $\mathbf u\times\mathbf v$ is perpendicular to both generating
directions and may be used as a normal vector of the plane. Reversing the order
only reverses the normal:
\[
\mathbf v\times\mathbf u=-(\mathbf u\times\mathbf v).
\]
Both choices describe the same plane.

\begin{derivationbox}[From the normal direction to an equation]
Let
\[
\mathbf n=\mathbf u\times\mathbf v.
\]
A point $P$ belongs to the plane exactly when $P-P_0$ is an allowed
combination. Hence
\[
\mathbf n\cdot(P-P_0)=0.
\]
For $\mathbf n=(a,b,c)$ this becomes
\[
a(x-x_0)+b(y-y_0)+c(z-z_0)=0,
\]
or, after collecting constants,
\[
ax+by+cz=d.
\]
\end{derivationbox}

\begin{theorembox}
The parametric and normal descriptions of the same plane are
\[
\boxed{
P=P_0+s\mathbf u+t\mathbf v
}
\]
and
\[
\boxed{
\mathbf n\cdot(P-P_0)=0
\quad\Longleftrightarrow\quad
ax+by+cz=d
}.
\]
\end{theorembox}

\begin{interpretationbox}[Two descriptions, one plane]
The parametric equation lists all permitted motions in the plane. The normal
equation states that the forbidden normal component of every in-plane
displacement must vanish.
\end{interpretationbox}

\subsection{A Plane Through Three Points}

\begin{derivationbox}[Extracting two independent directions]
Let $P,Q,R$ be three noncollinear points. Then
\[
\mathbf u=Q-P,
\qquad
\mathbf v=R-P
\]
are nonparallel and provide the two directions required to generate a plane.
Their cross product
\[
\mathbf n=(Q-P)\times(R-P)
\]
is nonzero and perpendicular to the plane.
\end{derivationbox}

\begin{theorembox}
The unique plane through three noncollinear points $P,Q,R$ is
\[
\boxed{
\Pi=\{P+s(Q-P)+t(R-P):s,t\in\mathbb R\}
}.
\]
A normal vector is
\[
\boxed{
\mathbf n=(Q-P)\times(R-P)
}.
\]
\end{theorembox}

\begin{examplebox}
For
\[
P=(1,0,0),\qquad Q=(0,1,0),\qquad R=(0,0,1),
\]
we obtain
\[
Q-P=(-1,1,0),\qquad R-P=(-1,0,1),
\]
and
\[
(Q-P)\times(R-P)=(1,1,1).
\]
Using the point $P$ gives
\[
(1,1,1)\cdot\bigl((x,y,z)-(1,0,0)\bigr)=0,
\]
so
\[
\boxed{x+y+z=1}.
\]
\end{examplebox}

\subsection{Relative Positions of Planes}

Let
\[
\Pi_1:\mathbf n_1\cdot P=d_1,
\qquad
\Pi_2:\mathbf n_2\cdot P=d_2.
\]

\begin{corebox}[Normals control relative position]
If the normals are parallel, the planes have the same orientation and are either
identical or distinct and parallel. If the normals are not parallel, the planes
intersect in a line.
\end{corebox}

\begin{theorembox}
The angle between planes is measured through their normals. In particular,
\[
\boxed{
\Pi_1\perp\Pi_2
\iff
\mathbf n_1\cdot\mathbf n_2=0
}.
\]
If the planes are nonparallel, a direction vector of their intersection line is
\[
\boxed{
\mathbf v=\mathbf n_1\times\mathbf n_2
}.
\]
\end{theorembox}

\begin{interpretationbox}[Finding the intersection line]
The cross product gives the direction because the line must be perpendicular to
both normals. A point on the line is then found by solving the two plane
equations simultaneously.
\end{interpretationbox}

\subsection{Distance from a Point to a Plane}

\begin{derivationbox}[Distance as a normal projection]
Let $P_0$ be a fixed point on the plane. The shortest displacement from $P$ to
the plane follows the normal direction. Hence the distance is the magnitude of
the scalar component of $P-P_0$ along $\mathbf n$:
\[
d(P,\Pi)=\frac{|\mathbf n\cdot(P-P_0)|}{\|\mathbf n\|}.
\]
\end{derivationbox}

\begin{theorembox}
For a plane $ax+by+cz=d$ and a point $P=(x_P,y_P,z_P)$,
\[
\boxed{
d(P,\Pi)=
\frac{|ax_P+by_P+cz_P-d|}{\sqrt{a^2+b^2+c^2}}
}.
\]
\end{theorembox}

\begin{interpretationbox}[What the formula measures]
The numerator is the signed normal displacement encoded by the plane equation;
the denominator converts the chosen normal vector to unit length. The formula
is an orthogonal projection, not an isolated computational rule.
\end{interpretationbox}

\begin{summarybox}[Line and plane from permitted directions]
A line is generated by one independent direction,
\[
P=P_0+t\mathbf v,
\]
whereas a plane is generated by two,
\[
P=P_0+s\mathbf u+t\mathbf v.
\]
The cross product of the two generating directions produces a normal, and the
dot product turns perpendicularity into the equation
\[
\mathbf n\cdot(P-P_0)=0.
\]
The parametric and normal forms therefore arise from the same geometric
construction.
\end{summarybox}
