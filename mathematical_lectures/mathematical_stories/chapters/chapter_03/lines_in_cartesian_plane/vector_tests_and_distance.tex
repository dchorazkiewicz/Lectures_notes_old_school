\subsection{Vector Tests Behind the Slope Rules}

\begin{theorembox}
Let two lines have nonzero direction vectors $\mathbf v_1$ and $\mathbf v_2$.
Their directions are parallel precisely when
\[
\boxed{
\mathbf v_1\parallel\mathbf v_2
\iff
\mathbf v_2=\lambda\mathbf v_1
\text{ for some }\lambda\neq0
}.
\]
They are perpendicular precisely when
\[
\boxed{
\mathbf v_1\perp\mathbf v_2
\iff
\mathbf v_1\cdot\mathbf v_2=0
}.
\]
\end{theorembox}

\begin{interpretationbox}[Why vector tests are more general]
These criteria do not require a slope and therefore include vertical directions
without any special convention. In three dimensions, parallel directions will
also be detected by a zero cross product.
\end{interpretationbox}

\begin{derivationbox}[Recovering the familiar slope rule]
For nonvertical lines with slopes $m_1,m_2$, choose direction vectors
\[
\mathbf v_1=(1,m_1),
\qquad
\mathbf v_2=(1,m_2).
\]
Then
\[
\mathbf v_1\cdot\mathbf v_2=1+m_1m_2.
\]
Therefore
\[
\mathbf v_1\perp\mathbf v_2
\iff
1+m_1m_2=0.
\]
\end{derivationbox}

\begin{theorembox}
For two nonvertical lines,
\[
\boxed{
L_1\perp L_2
\iff
m_1m_2=-1
}.
\]
This is the slope form of the dot-product criterion, not an independent rule.
\end{theorembox}

\subsection{The Nearest Point on a Line}

Let
\[
L=\{P_0+t\mathbf v:t\in\mathbb R\},
\qquad \mathbf v\neq\mathbf0,
\]
and let $P$ be any point.

\begin{derivationbox}[Projecting onto the allowed direction]
Decompose the displacement $P-P_0$ into a component parallel to $\mathbf v$ and
a component perpendicular to $\mathbf v$. The parallel component is
\[
\operatorname{proj}_{\mathbf v}(P-P_0)
=
\frac{(P-P_0)\cdot\mathbf v}{\mathbf v\cdot\mathbf v}\,\mathbf v.
\]
Starting from $P_0$ and moving by this component reaches
\[
Q=P_0+\operatorname{proj}_{\mathbf v}(P-P_0).
\]
\end{derivationbox}

\begin{theorembox}
The nearest point on $L$ to $P$ is
\[
\boxed{
Q=P_0+\operatorname{proj}_{\mathbf v}(P-P_0)
}.
\]
The remainder $P-Q$ is perpendicular to the line.
\end{theorembox}

\begin{center}
\begin{tikzpicture}[scale=0.95,>=stealth]
    \coordinate (P0) at (0.8,0.8);
    \coordinate (Q) at (3.7,2.1);
    \coordinate (P) at (2.8,4.1);
    \coordinate (R) at (5.4,2.86);
    \draw[subset] (0.1,0.49) -- (R) node[above right] {$L$};
    \draw[blue!70!black,line width=1.1pt,->] (P0) -- (Q)
        node[midway,below right] {$\operatorname{proj}_{\mathbf v}(P-P_0)$};
    \draw[orange!85!black,line width=1.1pt,->] (Q) -- (P)
        node[midway,right] {$P-Q$};
    \draw[very thick,->] (P0) -- (P)
        node[midway,above left] {$P-P_0$};
    \draw (Q) ++(-0.22,-0.10) -- ++(-0.12,0.27) -- ++(0.22,0.10);
    \fill[geometricSubsetColor] (P0) circle (2pt) node[below left] {$P_0$};
    \fill[geometricSubsetColor] (Q) circle (2pt) node[below right] {$Q$};
    \fill[geometricSubsetColor] (P) circle (2pt) node[above] {$P$};
\end{tikzpicture}
\end{center}

\begin{definitionbox}
The distance from $P$ to the line $L$ is
\[
\boxed{
d(P,L)
=
\left\|
(P-P_0)-\operatorname{proj}_{\mathbf v}(P-P_0)
\right\|
}.
\]
\end{definitionbox}

\begin{interpretationbox}[Why the perpendicular is shortest]
Every other route from $P$ to a point of the line contains the same perpendicular
component together with an additional component along the line. The orthogonal
component alone is therefore the shortest possible displacement.
\end{interpretationbox}

\subsection{Distance from the Normal Equation}

Suppose the line is written as
\[
ax+by=c,
\qquad (a,b)\neq(0,0),
\]
with normal vector $\mathbf n=(a,b)$.

\begin{derivationbox}[Reading the normal component]
For $P=(x_P,y_P)$ and any fixed $P_0$ on the line,
\[
\mathbf n\cdot(P-P_0)=ax_P+by_P-c.
\]
The signed scalar component in the normal direction is
\[
\frac{ax_P+by_P-c}{\sqrt{a^2+b^2}}.
\]
Distance is its absolute value.
\end{derivationbox}

\begin{theorembox}
For the line $ax+by=c$ and the point $P=(x_P,y_P)$,
\[
\boxed{
d(P,L)
=
\frac{|ax_P+by_P-c|}{\sqrt{a^2+b^2}}
}.
\]
\end{theorembox}

\begin{interpretationbox}[The formula is a projection]
The numerator measures signed displacement in the normal direction, and the
denominator normalises the normal vector. The formula is therefore the length
of a projection, not a detached rule to memorise.
\end{interpretationbox}
