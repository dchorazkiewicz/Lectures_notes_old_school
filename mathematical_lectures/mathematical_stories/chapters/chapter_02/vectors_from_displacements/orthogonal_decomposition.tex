\subsection{Orthogonal Decomposition and Its Consequences}

\begin{definitionbox}
For $\mathbf v\neq\mathbf0$, the orthogonal decomposition of $\mathbf u$ relative
to the direction of $\mathbf v$ is
\[
\boxed{
\mathbf u
=
\operatorname{proj}_{\mathbf v}\mathbf u
+
\left(\mathbf u-\operatorname{proj}_{\mathbf v}\mathbf u\right)
}.
\]
The first term is parallel to $\mathbf v$ and the second is perpendicular to
$\mathbf v$.
\end{definitionbox}

\begin{theorembox}
Because the two components are perpendicular,
\[
\boxed{
\|\mathbf u\|^2
=
\left\|\operatorname{proj}_{\mathbf v}\mathbf u\right\|^2
+
\left\|\mathbf u-\operatorname{proj}_{\mathbf v}\mathbf u\right\|^2
}.
\]
This is the Pythagorean theorem applied to vector components.
\end{theorembox}

\subsubsection*{Scalar component and vector projection}

\begin{definitionbox}
The signed scalar component of $\mathbf u$ in the direction of nonzero
$\mathbf v$ is
\[
\boxed{
\operatorname{comp}_{\mathbf v}\mathbf u
=
\frac{\mathbf u\cdot\mathbf v}{\|\mathbf v\|}
}.
\]
The corresponding vector projection is
\[
\boxed{
\operatorname{proj}_{\mathbf v}\mathbf u
=
\operatorname{comp}_{\mathbf v}\mathbf u
\frac{\mathbf v}{\|\mathbf v\|}
}.
\]
\end{definitionbox}

\begin{interpretationbox}[Scalar amount and vector displacement]
The scalar component measures signed length along the chosen direction. The
vector projection turns that signed amount back into an actual displacement by
multiplying it by the unit vector $\mathbf v/\|\mathbf v\|$.
\end{interpretationbox}

\subsubsection*{The Cauchy--Schwarz inequality}

\begin{proofbox}[Projection cannot be longer than the whole vector]
The length of the parallel component cannot exceed the length of $\mathbf u$:
\[
\left|\operatorname{comp}_{\mathbf v}\mathbf u\right|
\leq\|\mathbf u\|.
\]
Substituting the scalar-component formula gives
\[
\frac{|\mathbf u\cdot\mathbf v|}{\|\mathbf v\|}
\leq\|\mathbf u\|.
\]
Multiplying by $\|\mathbf v\|>0$ yields the inequality below.
\end{proofbox}

\begin{theorembox}
For all vectors $\mathbf u$ and $\mathbf v$,
\[
\boxed{
|\mathbf u\cdot\mathbf v|
\leq
\|\mathbf u\|\,\|\mathbf v\|
}.
\]
Consequently,
\[
-1\leq
\frac{\mathbf u\cdot\mathbf v}{\|\mathbf u\|\,\|\mathbf v\|}
\leq1
\]
for nonzero vectors, so the angle formula using $\arccos$ is meaningful.
\end{theorembox}

\subsubsection*{The triangle inequality}

\begin{proofbox}[The direct route is no longer than a broken route]
Using Cauchy--Schwarz,
\[
\begin{aligned}
\|\mathbf u+\mathbf v\|^2
&=\|\mathbf u\|^2+2\mathbf u\cdot\mathbf v+\|\mathbf v\|^2\\
&\leq\|\mathbf u\|^2+2\|\mathbf u\|\,\|\mathbf v\|+\|\mathbf v\|^2\\
&=(\|\mathbf u\|+\|\mathbf v\|)^2.
\end{aligned}
\]
Both sides are nonnegative, so taking square roots gives the desired result.
\end{proofbox}

\begin{theorembox}
For all vectors $\mathbf u$ and $\mathbf v$,
\[
\boxed{
\|\mathbf u+\mathbf v\|
\leq
\|\mathbf u\|+\|\mathbf v\|
}.
\]
\end{theorembox}

\begin{interpretationbox}[Geometric meaning]
The triangle inequality is the algebraic form of the Euclidean fact that the
straight segment between two points is no longer than a route that changes
direction at an intermediate point.
\end{interpretationbox}
