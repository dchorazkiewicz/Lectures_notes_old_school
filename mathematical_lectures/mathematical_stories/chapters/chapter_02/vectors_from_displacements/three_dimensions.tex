\subsection{From the Plane to Three-Dimensional Vectors}

\begin{corebox}[What changes in three dimensions]
Adding a third coordinate does not change the meaning of a vector. A vector is
still a displacement, now recording three independent coordinate changes:
\[
\boxed{\mathbf v=(v_1,v_2,v_3)}.
\]
Addition and scalar multiplication remain componentwise.
\end{corebox}

\begin{derivationbox}[Length in three dimensions]
Apply the Pythagorean theorem first to the first two coordinate changes and then
to the third. This gives
\[
\|\mathbf v\|^2=v_1^2+v_2^2+v_3^2.
\]
\end{derivationbox}

\begin{definitionbox}
For $\mathbf v=(v_1,v_2,v_3)$,
\[
\boxed{
\|\mathbf v\|=\sqrt{v_1^2+v_2^2+v_3^2}
}.
\]
For $\mathbf u=(u_1,u_2,u_3)$ and $\mathbf v=(v_1,v_2,v_3)$,
\[
\boxed{
\mathbf u\cdot\mathbf v
=u_1v_1+u_2v_2+u_3v_3
}.
\]
\end{definitionbox}

\begin{theorembox}
The geometric role of the dot product is unchanged:
\[
\mathbf u\cdot\mathbf v
=\|\mathbf u\|\,\|\mathbf v\|\cos\theta.
\]
Consequently,
\[
\boxed{
\mathbf u\perp\mathbf v
\iff
\mathbf u\cdot\mathbf v=0
}.
\]
\end{theorembox}

\subsubsection*{A second product available in three dimensions}

\begin{corebox}[The new geometric question]
Given two vectors $\mathbf u$ and $\mathbf v$, can we construct a vector
perpendicular to both of them?
\end{corebox}

\begin{definitionbox}
For
\[
\mathbf u=(u_1,u_2,u_3),
\qquad
\mathbf v=(v_1,v_2,v_3),
\]
the \emph{cross product} is
\[
\boxed{
\mathbf u\times\mathbf v
=
\bigl(
 u_2v_3-u_3v_2,
\;u_3v_1-u_1v_3,
\;u_1v_2-u_2v_1
\bigr)
}.
\]
\end{definitionbox}

\begin{proofbox}[The cross product is perpendicular to both inputs]
Direct substitution gives
\[
\begin{aligned}
(\mathbf u\times\mathbf v)\cdot\mathbf u
&=(u_2v_3-u_3v_2)u_1
 +(u_3v_1-u_1v_3)u_2
 +(u_1v_2-u_2v_1)u_3\\
&=0,
\end{aligned}
\]
and similarly
\[
(\mathbf u\times\mathbf v)\cdot\mathbf v=0.
\]
Therefore the cross product is perpendicular to both vectors.
\end{proofbox}

\begin{theorembox}
\[
\boxed{
(\mathbf u\times\mathbf v)\perp\mathbf u,
\qquad
(\mathbf u\times\mathbf v)\perp\mathbf v
}.
\]
Moreover, for all real $a,b$,
\[
\boxed{
(\mathbf u\times\mathbf v)\cdot(a\mathbf u+b\mathbf v)=0
}.
\]
Thus one cross product is perpendicular to every linear combination of its two
inputs.
\end{theorembox}

\begin{remarkbox}
We do not yet turn the family $a\mathbf u+b\mathbf v$ into a theory of planes.
That construction belongs to the next chapter. Here the cross product is
introduced only as a vector operation and its essential properties are proved.
\end{remarkbox}

\subsubsection*{The cross product detects parallelism}

\begin{proofbox}[Why zero cross product means parallel directions]
If $\mathbf v=\lambda\mathbf u$, substitution into the coordinate formula makes
every component of $\mathbf u\times\mathbf v$ vanish. Conversely, for nonzero
vectors, a zero cross product means that the two direction triples are linearly
dependent in no pair of coordinates and therefore lie on one common line.
\end{proofbox}

\begin{theorembox}
For nonzero vectors,
\[
\boxed{
\mathbf u\parallel\mathbf v
\iff
\mathbf u\times\mathbf v=\mathbf0
}.
\]
Together with the dot-product test,
\[
\boxed{
\begin{aligned}
\mathbf u\cdot\mathbf v=0
&\iff \mathbf u\perp\mathbf v,\\
\mathbf u\times\mathbf v=\mathbf0
&\iff \mathbf u\parallel\mathbf v.
\end{aligned}
}
\]
\end{theorembox}

\subsubsection*{Order and orientation}

\begin{definitionbox}
The direction of $\mathbf u\times\mathbf v$ is chosen by the right-hand rule:
curl the fingers of the right hand from $\mathbf u$ toward $\mathbf v$; the
thumb indicates the positive perpendicular direction.
\end{definitionbox}

\begin{theorembox}
Reversing the order reverses the cross product:
\[
\boxed{
\mathbf v\times\mathbf u
=-(\mathbf u\times\mathbf v)
}.
\]
Its length is
\[
\boxed{
\|\mathbf u\times\mathbf v\|
=\|\mathbf u\|\,\|\mathbf v\|\sin\theta
}.
\]
\end{theorembox}

\begin{interpretationbox}[What the magnitude records]
The factor $\sin\theta$ explains the parallelism test: it vanishes for
$\theta=0$ and $\theta=\pi$. The same magnitude will later measure the area of
the parallelogram generated by the two vectors.
\end{interpretationbox}

\begin{summarybox}[Dot product and cross product]
\[
\begin{array}{c|c|c}
\text{operation}&\text{output}&\text{main test}\\
\hline
\mathbf u\cdot\mathbf v&\text{number}&
\mathbf u\perp\mathbf v\iff\mathbf u\cdot\mathbf v=0\\
\mathbf u\times\mathbf v&\text{vector}&
\mathbf u\parallel\mathbf v\iff\mathbf u\times\mathbf v=\mathbf0
\end{array}
\]
The cross product is perpendicular to both inputs and to every linear
combination of them. These facts will become tools for lines and planes in the
following chapter.
\end{summarybox}
