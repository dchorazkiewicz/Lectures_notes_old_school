\subsection{A Vector Is a Displacement, Not a Place}

Let
\[
A=(x_1,y_1),
\qquad
B=(x_2,y_2).
\]
The point $A$ is a position and the point $B$ is another position. The arrow
from $A$ to $B$ asks a different question.

\begin{corebox}[The question answered by a vector]
What change of coordinates carries the starting point to the endpoint?
\end{corebox}

\begin{derivationbox}[Why endpoint minus starting point appears]
The required changes satisfy
\[
x_1+\Delta x=x_2,
\qquad
y_1+\Delta y=y_2.
\]
Solving gives
\[
\Delta x=x_2-x_1,
\qquad
\Delta y=y_2-y_1.
\]
Therefore
\[
\boxed{
\overrightarrow{AB}=(x_2-x_1,\,y_2-y_1)
}.
\]
This subtraction is forced by the requirement that the displacement carry $A$
to $B$.
\end{derivationbox}

\begin{definitionbox}
A vector in the Cartesian plane is a coordinate displacement
\[
\boxed{\mathbf v=(v_1,v_2)}.
\]
It instructs us to change the first coordinate by $v_1$ and the second
coordinate by $v_2$.
\end{definitionbox}

\begin{interpretationbox}[A vector is independent of its starting point]
If $P=(p_1,p_2)$ and the displacement is $\mathbf v=(v_1,v_2)$, then the endpoint
is
\[
Q=(p_1+v_1,\,p_2+v_2),
\]
and $\overrightarrow{PQ}=\mathbf v$. Two arrows drawn in different places
represent the same vector when they produce the same coordinate changes.
\end{interpretationbox}

\begin{definitionbox}
The zero vector
\[
\boxed{\mathbf0=(0,0)}
\]
means no displacement. The opposite vector
\[
\boxed{-\mathbf v=(-v_1,-v_2)}
\]
reverses the displacement.
\end{definitionbox}

\begin{theorembox}
If $\mathbf v$ carries $A$ to $B$, then $-\mathbf v$ carries $B$ back to $A$:
\[
\boxed{\overrightarrow{BA}=-\overrightarrow{AB}}.
\]
\end{theorembox}
