\section{Review Exercises}

\subsection{Elementary computational and graphical exercises}

\begin{enumerate}
\item Let $A=(1,2)$ and $B=(5,-1)$. Compute the free vector
$[\overrightarrow{AB}]$ and draw equal anchored representatives with initial
points $O=(0,0)$ and $P=(-2,3)$.

\item For
\[
\mathbf u=[3,-2],\qquad \mathbf v=[-1,4],
\]
compute $\mathbf u+\mathbf v$, $\mathbf u-\mathbf v$, $2\mathbf u$, and
$-3\mathbf v$. Draw $\mathbf u$, $\mathbf v$, and $\mathbf u+\mathbf v$ using
the parallelogram rule.

\item Compute $2\mathbf u-3\mathbf v$ for $\mathbf u=[1,2]$ and
$\mathbf v=[2,-1]$. Reproduce the same result by joining anchored
representatives head to tail.

\item Compute the lengths and unit vectors for
\[
\mathbf u=[3,4],\qquad
\mathbf v=[-5,12],\qquad
\mathbf w=[2,-2].
\]

\item For
\[
\mathbf u=[2,1],\qquad
\mathbf v=[1,-2],\qquad
\mathbf w=[4,2],
\]
compute the three dot products and decide which pairs are perpendicular or
parallel.

\item Find the angle between $\mathbf u=[1,0]$ and $\mathbf v=[1,1]$, and
between $\mathbf v=[1,1]$ and $\mathbf w=[-1,1]$.

\item Compute the projection of $\mathbf u=[4,3]$ onto $\mathbf v=[1,0]$ and
onto $\mathbf w=[1,1]$. In each case write $\mathbf u$ as a parallel component
plus a perpendicular remainder.

\item Let
\[
\mathbf b_1=[1,1],\qquad
\mathbf b_2=[1,-1],\qquad
\mathbf u=[4,2].
\]
Find $\alpha,\beta$ such that
\[
\mathbf u=\alpha\mathbf b_1+\beta\mathbf b_2,
\]
and verify the result by reconstruction. State the coordinates of $\mathbf u$
in the basis $B=(\mathbf b_1,\mathbf b_2)$.

\item Convert between Cartesian and polar coordinates:
\begin{enumerate}
\item $(\sqrt3,1)$ and $(-1,\sqrt3)$ to polar form;
\item $(r,\theta)=(4,\pi/6)$ and $(3,\pi)$ to Cartesian form.
\end{enumerate}
\end{enumerate}

\subsection{Development and reasoning exercises}

\begin{enumerate}
\item For
\[
\mathbf u=[5,1],\qquad \mathbf v=[2,1],
\]
compute the projection of $\mathbf u$ onto $\mathbf v$, the perpendicular
remainder, and verify by a dot product that the remainder is perpendicular to
$\mathbf v$. Draw the decomposition.

\item Choose two nonzero vectors $\mathbf u,\mathbf v$ with the same length but
with $\mathbf u+\mathbf v$ shorter than either vector. Compute the relevant
lengths and explain the geometric cancellation.

\item Find all values of $t$ for which
\[
\mathbf u=[1,t],\qquad \mathbf v=[2,-1]
\]
are perpendicular. Then find all values for which they are parallel. Interpret
both conditions geometrically.

\item Prove from the construction with anchored representatives, rather than
only from coordinates, that vector addition is associative. Illustrate the
argument with three concrete vectors.

\item Show that two nonzero vectors in the plane are parallel exactly when one
is a scalar multiple of the other. Connect the geometric statement with the
proportionality of their coordinates.

\item Explain how
\[
\mathbf u\cdot\mathbf v
=\lVert\mathbf u\rVert\lVert\mathbf v\rVert\cos\theta
\]
contains both length and angular information. Use one acute, one right, and one
obtuse example.

\item Compare Cartesian coordinates, coordinates in the basis
$B=(\mathbf b_1,\mathbf b_2)$ from the numerical example, and polar coordinates.
Give one concrete situation in which each description is useful.
\end{enumerate}
