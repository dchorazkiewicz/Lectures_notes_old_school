\subsection{Angles and Their Numerical Measure}

\begin{corebox}[Angle and angle measure are different objects]
An angle exists geometrically before it is assigned a number. We can compare,
copy, add, and bisect angles without degrees or radians. A numerical angle
measure requires an additional choice of unit.
\end{corebox}

\begin{definitionbox}
Let $r_1$ and $r_2$ be two rays with the same initial point $O$. The rays,
together with a choice of one of the two regions between them, determine an
angle $\alpha$ with vertex $O$.
\end{definitionbox}

\begin{remarkbox}
The angle is the chosen geometric region. It is neither the small arc drawn near
the vertex nor a number. The arc is only a graphical mark indicating which
region has been selected.
\end{remarkbox}

\begin{center}
\begin{tikzpicture}[scale=1.0]
    \coordinate (O) at (0,0);
    \coordinate (A) at (3.4,0);
    \coordinate (B) at (1.7,2.8);
    \fill[blue!8] (O) -- (A) arc[start angle=0,end angle=58.7,radius=3.4] -- cycle;
    \draw[axis] (O) -- (A) node[right] {$r_1$};
    \draw[axis] (O) -- (B) node[above] {$r_2$};
    \draw[subset] (0.9,0) arc[start angle=0,end angle=58.7,radius=0.9];
    \node[subset label] at (0.95,0.48) {$\alpha$};
    \fill (O) circle (2pt) node[below left] {$O$};
\end{tikzpicture}
\end{center}

\subsubsection*{Why circles give a consistent measure}

Draw a circle centred at $O$. The rays cut from the circle an arc corresponding
to the chosen angle.

\begin{theorembox}
For every Euclidean circle, the ratio of circumference $C$ to diameter $D$ is
the same constant:
\[
\frac{C}{D}=\pi.
\]
Since $D=2r$, every circle of radius $r$ has circumference
\[
\boxed{C=2\pi r}.
\]
\end{theorembox}

\begin{proofbox}[Why the quotient $s/r$ is independent of the circle]
Similarity implies that enlarging a circle by a factor $\lambda$ multiplies its
diameter, circumference, and corresponding arc lengths by $\lambda$. If the
same rays cut arc lengths $s_1,s_2$ from circles of radii $r_1,r_2$, then
\[
\frac{s_1}{s_2}=\frac{r_1}{r_2},
\qquad\text{hence}\qquad
\frac{s_1}{r_1}=\frac{s_2}{r_2}.
\]
The quotient therefore depends only on the angle.
\end{proofbox}

\begin{definitionbox}
Let $s$ be the length of the selected arc on a circle of radius $r$ centred at
the vertex. The radian measure of $\alpha$ is
\[
\boxed{m_{\mathrm{rad}}(\alpha)=\frac{s}{r}}.
\]
\end{definitionbox}

\begin{derivationbox}[Full turn, half-turn, and right angle]
For a full turn, $s=2\pi r$, so
\[
m_{\mathrm{rad}}(\text{full turn})=2\pi.
\]
A half-turn therefore has measure $\pi$, and a quarter-turn has measure $\pi/2$.
\end{derivationbox}

\begin{definitionbox}
Degrees provide a second numerical scale. A full turn is assigned $360^\circ$,
so
\[
2\pi\text{ radians}=360^\circ,
\qquad
\boxed{
m_{\deg}(\alpha)=\frac{180^\circ}{\pi}\,m_{\mathrm{rad}}(\alpha)
}.
\]
\end{definitionbox}

\begin{examplebox}
A constructible right angle has the two numerical descriptions
\[
m_{\mathrm{rad}}(\text{right angle})=\frac{\pi}{2},
\qquad
m_{\deg}(\text{right angle})=90^\circ.
\]
The geometric angle is the same object; only the chosen numerical scale changes.
\end{examplebox}

\subsubsection*{Sine and cosine as functions of a number}

The trigonometric ratios were first defined for acute geometric angles. We now
extend them to a full turn. For $t\in[0,2\pi)$, let $\beta\in[0,\pi/2]$ be the
acute reference angle between the direction of $t$ and the nearest horizontal
ray. The magnitudes $\sin\beta$ and $\cos\beta$ come from the right-triangle
definition; their signs are determined by the quadrant:
\[
\begin{array}{c|cc}
&\cos t&\sin t\\ \hline
0\le t\le \pi/2&+\cos\beta&+\sin\beta\\
\pi/2\le t\le \pi&-\cos\beta&+\sin\beta\\
\pi\le t\le 3\pi/2&-\cos\beta&-\sin\beta\\
3\pi/2\le t<2\pi&+\cos\beta&-\sin\beta
\end{array}
\]

\begin{definitionbox}
For arbitrary real $t$, define sine and cosine periodically:
\[
\boxed{
\sin(t+2\pi k)=\sin t,
\qquad
\cos(t+2\pi k)=\cos t,
\qquad k\in\mathbb Z
}.
\]
\end{definitionbox}

\begin{interpretationbox}[From a geometric angle to a numerical input]
If $t=m_{\mathrm{rad}}(\alpha)$, then $\cos t$ is the same value as the
geometric cosine $\cos\alpha$. The notation $\cos t$ therefore treats the
numerical measure as an input while preserving the earlier geometric meaning.
\end{interpretationbox}

\subsubsection*{The Euclidean law of cosines}

Directly measuring an arc is often difficult. A triangle provides another way
to recover the measure of an angle. Choose points
\[
A\in r_1\setminus\{O\},
\qquad
B\in r_2\setminus\{O\}.
\]
The points form the Euclidean triangle $OAB$.

\begin{center}
\begin{tikzpicture}[scale=0.92]
    \coordinate (O) at (0,0);
    \coordinate (A) at (4.4,0);
    \coordinate (B) at (1.55,3.0);

    \fill[blue!6] (O) -- (1.0,0)
        arc[start angle=0,end angle=62.7,radius=1.0] -- cycle;

    \draw[subset] (O) -- (A) node[midway,below=5pt] {$|OA|$};
    \draw[subset] (O) -- (B) node[midway,left=7pt] {$|OB|$};
    \draw[subset] (A) -- (B) node[midway,above right=3pt] {$|AB|$};

    \draw[subset] (0.78,0)
        arc[start angle=0,end angle=62.7,radius=0.78];
    \node[subset label] at (0.88,0.42) {$\alpha$};

    \fill (O) circle (2.1pt) node[below left] {$O$};
    \fill (A) circle (2.1pt) node[below right] {$A$};
    \fill (B) circle (2.1pt) node[above] {$B$};
\end{tikzpicture}
\end{center}

\begin{theorembox}
If $\alpha=\angle AOB$, then
\[
\boxed{
|AB|^2=|OA|^2+|OB|^2-2|OA|\,|OB|\cos\alpha
}.
\]
Equivalently, the last factor may be written as
$\cos(m_{\mathrm{rad}}(\alpha))$.
\end{theorembox}

\begin{derivationbox}[Recovering the angle from side lengths]
Solving the law of cosines for the angle measure gives
\[
\boxed{
m_{\mathrm{rad}}(\alpha)
=
\arccos\left(
\frac{|OA|^2+|OB|^2-|AB|^2}{2|OA|\,|OB|}
\right)
}.
\]
Later, coordinates will provide an arithmetic method for calculating the three
Euclidean lengths in this formula.
\end{derivationbox}

\begin{remarkbox}
At this point no vector has been introduced. We have only Euclidean objects,
postulates, constructions, similarity, trigonometric ratios, numerical angle
measure, and the Euclidean law of cosines. Vectors will appear only after the
Cartesian coordinate system has been constructed.
\end{remarkbox}

\begin{summarybox}[What numerical measurement adds]
The angle is geometric; radians and degrees are numerical descriptions of it.
Similarity makes the radian quotient $s/r$ independent of scale, while the law
of cosines allows the same angle to be recovered from three side lengths.
\end{summarybox}
