\section{Points, Coordinates, and the Euclidean Plane}

Before using coordinates for geometry, we must distinguish four different
objects:

\begin{enumerate}
    \item the set of ordered pairs of real numbers;
    \item a coordinate interpretation of those pairs;
    \item a rule that measures distance between pairs;
    \item geometric notions, such as length, circle, angle, and perpendicularity,
          defined relative to the chosen structure.
\end{enumerate}

These objects are related, but they are not identical. In particular, the bare
set $\mathbb{R}^2$ contains neither distances nor angles.

\subsection{The Set of Ordered Pairs}

\begin{definitionbox}
The Cartesian product of two copies of the real line is
\[
\mathbb{R}^2
=
\mathbb{R}\times\mathbb{R}
=
\{(x,y):x\in\mathbb{R},\ y\in\mathbb{R}\}.
\]
Its elements are ordered pairs of real numbers.
\end{definitionbox}

The word \emph{ordered} is essential:
\[
(2,5)\neq(5,2).
\]
At this stage, $(x,y)$ is only an ordered pair. Nothing in the set
$\mathbb{R}^2$ alone says that $x$ is horizontal position, that $y$ is vertical
position, or that the two directions are perpendicular.

\begin{remarkbox}
A set specifies which objects exist. It does not, by itself, specify how far
apart they are, which curves are circles, or what an angle means.
\end{remarkbox}

\subsection{A Coordinate Interpretation}

To use $\mathbb{R}^2$ as a model of a plane, we choose two real number lines,
place one horizontally and the other vertically, and make their zero points
coincide.

\begin{center}
\begin{tikzpicture}[scale=0.9]
    \draw[axis] (-4.5,0) -- (4.7,0) node[right] {$x$};
    \draw[axis] (0,-3.5) -- (0,3.7) node[above] {$y$};
    \foreach \x in {-4,-3,-2,-1,1,2,3,4} {
        \draw (\x,0.08) -- (\x,-0.08) node[below=3pt] {\small $\x$};
    }
    \foreach \y in {-3,-2,-1,1,2,3} {
        \draw (0.08,\y) -- (-0.08,\y) node[left=3pt] {\small $\y$};
    }
    \node[below left=2pt] at (0,0) {$0$};
\end{tikzpicture}
\end{center}

The horizontal copy is the $x$-axis, the vertical copy is the $y$-axis, and
their common zero is the origin
\[
O=(0,0).
\]

\begin{definitionbox}
Under this coordinate interpretation, the ordered pair
\[
P=(x,y)
\]
labels the point reached by taking position $x$ on the horizontal axis and
position $y$ on the vertical axis. The numbers $x$ and $y$ are called the
coordinates of $P$.
\end{definitionbox}

This interpretation gives a correspondence
\[
\text{coordinate point}
\quad\longleftrightarrow\quad
\text{ordered pair in }\mathbb{R}^2.
\]
It still does not define distance or angle.

\subsection{Distance: Additional Geometric Structure}

Let
\[
P=(x_1,y_1),
\qquad
Q=(x_2,y_2).
\]
Write
\[
\Delta x=x_2-x_1,
\qquad
\Delta y=y_2-y_1.
\]
The Pythagorean relation motivates the Euclidean distance.

\begin{definitionbox}
The Euclidean distance between $P$ and $Q$ is
\[
d_E(P,Q)
=
\sqrt{(x_2-x_1)^2+(y_2-y_1)^2}.
\]
\end{definitionbox}

\begin{center}
\begin{tikzpicture}[scale=0.9]
    \draw[axis] (-0.8,0) -- (5.2,0) node[right] {$x$};
    \draw[axis] (0,-0.8) -- (0,4.5) node[above] {$y$};
    \coordinate (P) at (1,1);
    \coordinate (Q) at (4,3.5);
    \coordinate (R) at (4,1);
    \draw[subset] (P) -- (Q);
    \draw[helper] (P) -- (R) node[midway,below] {$|\Delta x|$};
    \draw[helper] (R) -- (Q) node[midway,right] {$|\Delta y|$};
    \draw (3.72,1) -- (3.72,1.28) -- (4,1.28);
    \fill (P) circle (2pt) node[below left] {$P$};
    \fill (Q) circle (2pt) node[above right] {$Q$};
    \node[subset label,above left] at (2.5,2.35) {$d_E(P,Q)$};
\end{tikzpicture}
\end{center}

The defining relation is
\[
\boxed{d_E(P,Q)^2=(\Delta x)^2+(\Delta y)^2}.
\]

\begin{theorembox}
For all $P,Q,R\in\mathbb{R}^2$, the function $d_E$ satisfies
\[
\begin{aligned}
d_E(P,Q)&\geq 0,\\
d_E(P,Q)=0&\iff P=Q,\\
d_E(P,Q)&=d_E(Q,P),\\
d_E(P,R)&\leq d_E(P,Q)+d_E(Q,R).
\end{aligned}
\]
Therefore $d_E$ is a metric on $\mathbb{R}^2$.
\end{theorembox}

\subsection{The Euclidean Plane}

\begin{definitionbox}
The standard Euclidean plane in coordinates is
\[
\boxed{\bigl(\mathbb{R}^2,d_E\bigr)}.
\]
It consists of the set $\mathbb{R}^2$ equipped with the Euclidean metric $d_E$.
\end{definitionbox}

Thus
\[
\boxed{\mathbb{R}^2\text{ is a set}}
\qquad\text{whereas}\qquad
\boxed{(\mathbb{R}^2,d_E)\text{ is a metric space}}.
\]

\subsection{Angles Depend on Geometric Structure}

There is no angle attached to three points merely because the points belong to
$\mathbb{R}^2$. An angle must be defined relative to additional geometric
structure.

For the Euclidean metric, the angle can be reconstructed from distances. Let
$A,P,B\in\mathbb{R}^2$, with $A\neq P$ and $B\neq P$. The three Euclidean
distances
\[
d_E(A,P),
\qquad
d_E(B,P),
\qquad
d_E(A,B)
\]
determine the Euclidean triangle $APB$ up to rigid motion.

\begin{definitionbox}
The Euclidean angle $\angle APB\in[0,\pi]$ is defined by
\[
\cos\angle APB
=
\frac{
 d_E(A,P)^2+d_E(B,P)^2-d_E(A,B)^2
}{
 2d_E(A,P)d_E(B,P)
}.
\]
Equivalently,
\[
\angle APB
=
\arccos\left(
\frac{
 d_E(A,P)^2+d_E(B,P)^2-d_E(A,B)^2
}{
 2d_E(A,P)d_E(B,P)
}
\right).
\]
\end{definitionbox}

This is the law of cosines used as a definition of the Euclidean angle in
terms of the Euclidean metric.

\begin{theorembox}
The angle $\angle APB$ is a right angle precisely when
\[
\angle APB=\frac{\pi}{2}.
\]
Since $\cos(\pi/2)=0$, this is equivalent to
\[
\boxed{
d_E(A,B)^2
=
d_E(A,P)^2+d_E(B,P)^2
}.
\]
Thus Euclidean perpendicularity can be expressed entirely through Euclidean
distances.
\end{theorembox}

\begin{remarkbox}
The statement above is special to Euclidean geometry. An arbitrary metric does
not automatically determine a canonical notion of angle. If one replaces
$d_E$ by another metric and inserts its distances into the Euclidean cosine
formula, the resulting quantity need not have the geometric meaning of an
angle and may not define a coherent angle theory at all.

Therefore one must not say that every metric simply produces its own angles.
The correct statement is that the Euclidean metric determines the Euclidean
angles because it has the special Pythagorean structure encoded in the law of
cosines.
\end{remarkbox}

The dependence is therefore
\[
\boxed{
\text{Euclidean metric}
\quad\longrightarrow\quad
\text{Euclidean lengths and angles}
\quad\longrightarrow\quad
\text{Euclidean perpendicularity}
}.
\]

\subsection{Other Metrics on the Same Set}

Other metrics may be placed on the same set $\mathbb{R}^2$. For example,
\[
d_1(P,Q)=|x_2-x_1|+|y_2-y_1|
\]
and
\[
d_\infty(P,Q)=\max\{|x_2-x_1|,|y_2-y_1|\}.
\]
They produce different distances and different metric circles.

\begin{remarkbox}
Changing the metric changes the metric geometry. It does not follow that the
new metric carries a natural angle satisfying the Euclidean law of cosines.
Distance and angle are related in Euclidean geometry, but they are not
universally interchangeable concepts in every metric space.
\end{remarkbox}

\subsection{Coordinate Conditions and Subsets}

Once the ambient set has been specified, a condition on $x$ and $y$ selects a
subset of $\mathbb{R}^2$. For example,
\[
y=2
\]
is a condition, whereas
\[
L=\{(x,y)\in\mathbb{R}^2:y=2\}
\]
defines the complete set of pairs satisfying it.

\begin{definitionbox}
The expression
\[
\{(x,y)\in\mathbb{R}^2:\mathcal{C}(x,y)\}
\]
denotes the set of all ordered pairs in $\mathbb{R}^2$ satisfying the condition
$\mathcal{C}(x,y)$.
\end{definitionbox}

The colon is read as ``such that.'' A vertical bar may be used instead:
\[
\{(x,y)\in\mathbb{R}^2:y=2\}
=
\{(x,y)\in\mathbb{R}^2\mid y=2\}.
\]

Under the coordinate interpretation, $L$ is a horizontal line.

\begin{center}
\begin{tikzpicture}[scale=0.8]
    \draw[axis] (-4.2,0) -- (4.4,0) node[right] {$x$};
    \draw[axis] (0,-1.3) -- (0,3.8) node[above] {$y$};
    \draw[subset] (-4,2) -- (4,2);
    \node[subset label,right] at (4,2) {$L$};
    \draw (0.08,2) -- (-0.08,2) node[left=3pt] {$2$};
\end{tikzpicture}
\end{center}

The coordinate axes are the subsets
\[
X=\{(x,y)\in\mathbb{R}^2:y=0\},
\qquad
Y=\{(x,y)\in\mathbb{R}^2:x=0\},
\]
and
\[
X\cap Y=\{(0,0)\}.
\]

\subsection{A Metric Circle}

\begin{definitionbox}
For a centre $O\in\mathbb{R}^2$ and radius $R>0$, the Euclidean circle is
\[
C_E(O,R)
=
\{P\in\mathbb{R}^2:d_E(P,O)=R\}.
\]
\end{definitionbox}

For $O=(0,0)$ and $R=2$,
\[
d_E((x,y),O)=2
\]
is equivalent to
\[
x^2+y^2=4.
\]
Therefore
\[
C_E(O,2)
=
\{(x,y)\in\mathbb{R}^2:x^2+y^2=4\}.
\]

\begin{center}
\begin{tikzpicture}[scale=0.9]
    \draw[axis] (-3.2,0) -- (3.4,0) node[right] {$x$};
    \draw[axis] (0,-3.2) -- (0,3.4) node[above] {$y$};
    \draw[subset] (0,0) circle (2);
    \fill (0,0) circle (1.7pt) node[below left] {$O$};
    \node[subset label,above right] at (1.4,1.4) {$C_E(O,2)$};
\end{tikzpicture}
\end{center}

\begin{summarybox}
The logical order is
\[
\mathbb{R}^2
\longrightarrow
\text{coordinate interpretation}
\longrightarrow
d_E
\longrightarrow
(\mathbb{R}^2,d_E)
\longrightarrow
\text{Euclidean lengths, circles, angles, and perpendicularity}.
\]
None of the later geometric notions belongs to the bare set
$\mathbb{R}^2$ without additional structure.
\end{summarybox}
