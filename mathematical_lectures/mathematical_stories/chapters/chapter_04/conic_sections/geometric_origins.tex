\section{The Classical Conics in Cartesian Coordinates}

The previous chapter studied lines and planes: geometric sets produced by one or
two independent directions. We now meet curves whose shape is controlled by
distance. Their equations contain squares because Euclidean distance contains
squares.

\begin{corebox}[The plan of the chapter]
We first recall the four classical conics in Cartesian coordinates. We then
introduce polar coordinates, place the pole at a focus, and show that ellipse,
parabola, and hyperbola are governed by one equation and one parameter: the
eccentricity.
\end{corebox}

\subsection{Circle: one centre and one radius}

A circle with centre $C=(h,k)$ and radius $R>0$ is the set of points satisfying
\[
d(P,C)=R.
\]
Therefore
\[
\boxed{(x-h)^2+(y-k)^2=R^2}.
\]

\begin{center}
\begin{tikzpicture}[scale=0.9,>=stealth]
    \draw[axis,->] (-3.8,0)--(4.2,0) node[right] {$x$};
    \draw[axis,->] (0,-3.2)--(0,3.6) node[above] {$y$};
    \coordinate (C) at (0.8,0.5);
    \coordinate (P) at ({0.8+2.2*cos(38)},{0.5+2.2*sin(38)});
    \draw[subset] (C) circle (2.2);
    \draw[blue!70!black,line width=1.1pt] (C)--(P)
        node[midway,above left] {$R$};
    \fill[geometricSubsetColor] (C) circle (2.2pt) node[below left] {$C$};
    \fill[geometricSubsetColor] (P) circle (2.2pt) node[above right] {$P$};
\end{tikzpicture}
\end{center}

\begin{interpretationbox}[Equation and motion]
The Cartesian equation tests membership. A parameter can instead move a point
around the curve:
\[
\boxed{
\begin{aligned}
x&=h+R\cos t,\\
y&=k+R\sin t.
\end{aligned}}
\]
As $t$ runs from $0$ to $2\pi$, the point completes one circuit.
\end{interpretationbox}

\subsection{Ellipse: a stretched circle and a two-focus curve}

The standard ellipse centred at the origin is
\[
\boxed{\frac{x^2}{a^2}+\frac{y^2}{b^2}=1},
\qquad a\geq b>0.
\]
It can be viewed as a circle stretched differently in two perpendicular
directions.

\begin{center}
\begin{tikzpicture}[scale=0.85,>=stealth]
    \draw[axis,->] (-4.5,0)--(4.7,0) node[right] {$x$};
    \draw[axis,->] (0,-3.0)--(0,3.2) node[above] {$y$};
    \draw[subset] (0,0) ellipse (3.3 and 1.9);
    \draw[helper] (0,0)--(3.3,0) node[midway,below] {$a$};
    \draw[helper] (0,0)--(0,1.9) node[midway,left] {$b$};
    \coordinate (F1) at (-2.7,0);
    \coordinate (F2) at (2.7,0);
    \coordinate (P) at (1.5,1.7);
    \draw[blue!70!black,line width=1pt] (F1)--(P);
    \draw[orange!85!black,line width=1pt] (P)--(F2);
    \fill[geometricSubsetColor] (F1) circle (2pt) node[below] {$F_1$};
    \fill[geometricSubsetColor] (F2) circle (2pt) node[below] {$F_2$};
    \fill[geometricSubsetColor] (P) circle (2pt) node[above right] {$P$};
\end{tikzpicture}
\end{center}

Let
\[
c^2=a^2-b^2.
\]
Then the foci are $F_1=(-c,0)$ and $F_2=(c,0)$, and the same ellipse is described
geometrically by
\[
\boxed{d(P,F_1)+d(P,F_2)=2a}.
\]

\begin{interpretationbox}[A useful parameterisation]
The ellipse has the parameterisation
\[
\boxed{
\begin{aligned}
x&=a\cos t,\\
y&=b\sin t.
\end{aligned}}
\]
It automatically satisfies the Cartesian equation because
$\cos^2t+\sin^2t=1$. The parameter $t$ is convenient, but it is not generally the
polar angle measured from a focus.
\end{interpretationbox}

\subsection{Parabola: equal distance from a focus and a directrix}

Fix a focus $F=(p,0)$ and the directrix $x=-p$, where $p>0$. A point $P$ belongs
to the parabola when
\[
\boxed{d(P,F)=d(P,D)}.
\]
This condition gives the standard equation
\[
\boxed{y^2=4px}.
\]

\begin{center}
\begin{tikzpicture}[scale=0.82,>=stealth]
    \draw[axis,->] (-2.5,0)--(6.0,0) node[right] {$x$};
    \draw[axis,->] (0,-4.0)--(0,4.2) node[above] {$y$};
    \draw[gray,thick] (-1.2,-3.7)--(-1.2,3.7) node[above] {directrix};
    \draw[subset,domain=-3.3:3.3,samples=100]
        plot ({0.42*\x*\x},{\x});
    \coordinate (F) at (1.2,0);
    \coordinate (P) at (2.7,2.55);
    \coordinate (H) at (-1.2,2.55);
    \draw[blue!70!black,line width=1pt] (F)--(P)
        node[midway,below right] {$d(P,F)$};
    \draw[orange!85!black,line width=1pt] (P)--(H)
        node[midway,above] {$d(P,D)$};
    \fill[geometricSubsetColor] (F) circle (2pt) node[below] {$F$};
    \fill[geometricSubsetColor] (P) circle (2pt) node[above right] {$P$};
\end{tikzpicture}
\end{center}

\begin{interpretationbox}[An algebraic parameter]
The substitution
\[
\boxed{x=pt^2,\qquad y=2pt}
\]
traces the parabola because $y^2=4p^2t^2=4px$. Unlike the circle and ellipse,
this curve is not closed: the parameter can move arbitrarily far from the
vertex.
\end{interpretationbox}

\subsection{Hyperbola: a fixed difference of distances}

For $a>0$, $b>0$, the standard hyperbola is
\[
\boxed{\frac{x^2}{a^2}-\frac{y^2}{b^2}=1}.
\]
Let
\[
c^2=a^2+b^2.
\]
Then the foci are $F_1=(-c,0)$ and $F_2=(c,0)$, and the geometric condition is
\[
\boxed{|d(P,F_1)-d(P,F_2)|=2a}.
\]

\begin{center}
\begin{tikzpicture}[scale=0.76,>=stealth]
    \draw[axis,->] (-5.2,0)--(5.4,0) node[right] {$x$};
    \draw[axis,->] (0,-4.0)--(0,4.2) node[above] {$y$};
    \draw[helper,dashed] (-4.8,-3.2)--(4.8,3.2);
    \draw[helper,dashed] (-4.8,3.2)--(4.8,-3.2);
    \draw[subset,domain=1.55:4.7,samples=100]
        plot (\x,{1.15*sqrt(\x*\x/2.25-1)});
    \draw[subset,domain=1.55:4.7,samples=100]
        plot (\x,{-1.15*sqrt(\x*\x/2.25-1)});
    \draw[subset,domain=-4.7:-1.55,samples=100]
        plot (\x,{1.15*sqrt(\x*\x/2.25-1)});
    \draw[subset,domain=-4.7:-1.55,samples=100]
        plot (\x,{-1.15*sqrt(\x*\x/2.25-1)});
    \fill[geometricSubsetColor] (-2.7,0) circle (2pt) node[below] {$F_1$};
    \fill[geometricSubsetColor] (2.7,0) circle (2pt) node[below] {$F_2$};
    \node at (3.9,3.0) {asymptote};
\end{tikzpicture}
\end{center}

Its asymptotes are
\[
\boxed{y=\pm\frac ba x}.
\]
They describe the directions approached by the branches far from the centre.

\begin{interpretationbox}[Closed and open conics]
The circle and ellipse are bounded and closed. The parabola and hyperbola are
unbounded. This distinction will later reappear in mechanics as the difference
between bound and escape trajectories.
\end{interpretationbox}

\begin{summarybox}[Four familiar curves, one geometric programme]
The Cartesian equations make the curves calculable. The focus and directrix
make their geometry visible. Parameterisations describe motion along them. The
next step is to choose coordinates centred at a focus, where all three
noncircular conics can be described by one formula.
\end{summarybox}
