\subsection{Long-term behavior of discrete values}

\begin{definitionbox}[Limit of a sequence]
We write
\[
\lim_{n\to\infty}a_n=L
\]
when the terms $a_n$ become and remain arbitrarily close to $L$ for sufficiently large $n$.
\end{definitionbox}

\begin{examplebox}[A basic convergent sequence]
For
\[
a_n=\frac1n,
\]
the terms are positive and decrease toward zero. Thus
\[
\lim_{n\to\infty}\frac1n=0.
\]
\end{examplebox}

\begin{examplebox}[Dominant powers]
For
\[
a_n=\frac{2n+1}{n+3},
\]
divide numerator and denominator by $n$:
\[
a_n=\frac{2+1/n}{1+3/n}\to2.
\]
\end{examplebox}

\begin{remarkbox}
A sequence may fail to converge by growing without bound or oscillating without approaching one value. The existence of infinitely many terms does not guarantee a finite limit.
\end{remarkbox}
