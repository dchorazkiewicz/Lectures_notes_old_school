\section{Review Exercises}

\subsection{Elementary computational exercises}

\begin{enumerate}
\item Compute the first six terms and determine the limit of
\[
a_n=\frac1n,
\qquad b_n=2-\frac1n,
\qquad c_n=\left(\frac12\right)^n.
\]
State monotonicity and boundedness.

\item For $a_n=1/n$, find $N$ such that $n\ge N$ implies $|a_n|<0.01$. Then give a suitable formula for arbitrary $\varepsilon>0$.

\item Compute
\[
\lim_{n\to\infty}\frac{3n+1}{n+4},
\qquad
\lim_{n\to\infty}\frac{2n^2-n}{5n^2+3},
\qquad
\lim_{n\to\infty}\frac4{n^2+1}.
\]

\item For $a_n=1-2^{-n}$, show that the sequence is increasing and bounded above, then calculate its limit.

\item For $f(x)=2x+1$, $a=3$, and $L=7$, find $\delta$ in terms of $\varepsilon$ so that
\[
0<|x-3|<\delta\Longrightarrow |f(x)-7|<\varepsilon.
\]

\item Let $x_n=3+(-1)^n/n$. Compute $f(x_n)$ for $f(x)=2x+1$ and use the result to illustrate Heine's theorem.

\item Compute by factoring and cancellation:
\[
\lim_{x\to4}\frac{x^2-2x-8}{x^2-9x+20},
\qquad
\lim_{x\to2}\frac{x^2-4}{x-2}.
\]

\item Compute
\[
\lim_{x\to0}\frac{\sqrt{1+x}-1}{x}
\]
using the conjugate and
\[
\lim_{x\to0}\frac{\sin x}{x}
\]
using the standard limit.

\item For
\[
f(x)=\begin{cases}
x+1,&x<0,\\
2-x,&x>0,
\end{cases}
\]
compute the one-sided limits at $0$ and decide whether a value of $f(0)$ can make the function continuous.

\item Find the vertical and horizontal asymptotes of
\[
f(x)=\frac1{x-2},
\qquad
g(x)=\frac{3x^2+x}{x^2+4}
\]
from the relevant limits.
\end{enumerate}

\subsection{Development and reasoning exercises}

\begin{enumerate}
\item Prove from the $\varepsilon$--$N$ definition that
\[
\frac{3}{2n+1}\longrightarrow0.
\]
Give an explicit $N(\varepsilon)$ and explain every inequality.

\item Prove that
\[
\frac{2n+1}{n+3}\longrightarrow2
\]
using the formal definition, not only dominant terms.

\item For $f(x)=x^2$, prove
\[
\lim_{x\to2}x^2=4
\]
with an $\varepsilon$--$\delta$ argument. First control $|x+2|$ by restricting $\delta$, then choose a complete formula for $\delta(\varepsilon)$.

\item Use Heine's theorem to disprove the existence of
\[
\lim_{x\to0}\frac{x}{|x|}.
\]
Construct two sequences, compute both output sequences, and state precisely where the common-limit requirement fails.

\item Investigate the Cauchy condition for
\[
f(x)=\sin(1/x),\qquad x\ne0.
\]
Choose nearby sequences whose function values remain separated and explain why no finite limit exists at $0$.

\item Determine constants $a,b$ so that
\[
f(x)=\begin{cases}
ax+b,&x<1,\\
x^2,&x\ge1
\end{cases}
\]
is continuous at $1$ and also satisfies $f(0)=2$. Show the two conditions separately.

\item Analyse
\[
f(x)=\frac{x^2-1}{x-1}
\]
at $x=1$: calculate the nearby limit, identify the discontinuity, define a continuous extension, and explain why changing one point does not alter the limit.

\item Use the intermediate value theorem to prove that $x^3-x-1=0$ has a solution in $(1,2)$. Then use monotonicity on a suitable interval to discuss uniqueness.

\item Compare the $\varepsilon$--$\delta$, Heine, and Cauchy descriptions of a function limit. Apply all three viewpoints to the same simple linear function and identify what each formulation controls.

\item Give a complete limit-analysis strategy: determine the domain, inspect direct substitution, simplify when necessary, compare one-sided behaviour, use sequences to test failure, identify asymptotes, and decide continuity. Apply the strategy to one rational or piecewise function.
\end{enumerate}
