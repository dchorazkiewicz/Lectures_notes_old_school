\subsection{The Adjugate Inverse Formula}

\begin{theorembox}[Adjugate identity]
For every square matrix,
\[
A\operatorname{adj}(A)=(\det A)I.
\]
If $\det A\neq0$, then
\[
\boxed{A^{-1}=\frac1{\det A}\operatorname{adj}(A)}.
\]
\end{theorembox}

\begin{remarkbox}[Proof status]
The identity comes from Laplace expansion: diagonal entries of $A\operatorname{adj}(A)$ reproduce $\det A$, while off-diagonal entries cancel because they correspond to determinants with repeated rows or columns. This mechanism is the point retained here; a full index-by-index proof is not developed.
\end{remarkbox}

\begin{derivationbox}[Why the identity gives an inverse]
Starting from
\[
A\operatorname{adj}(A)=(\det A)I,
\]
divide by the nonzero scalar $\det A$:
\[
A\left(\frac1{\det A}\operatorname{adj}(A)\right)=I.
\]
The matrix in parentheses is therefore the inverse.
\end{derivationbox}

\begin{interpretationbox}[Determinants and cofactors build the inverse]
The same smaller determinants used in Laplace expansion assemble the matrix that reverses $A$. A zero determinant prevents the final division and records the loss of invertibility.
\end{interpretationbox}
