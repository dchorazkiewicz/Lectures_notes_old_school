\section{Problem Set 8: Sequences and Functions}

\begin{enumerate}
\item Decide which of the following assignments define functions:
\begin{enumerate}
\item each real number \(x\) is assigned \(x^2\);
\item each positive number \(x\) is assigned both \(\sqrt{x}\) and \(-\sqrt{x}\);
\item each student in a class is assigned the student's date of birth.
\end{enumerate}
Explain each decision.

\item For
\[
f:\mathbb{R}\to\mathbb{R},
\qquad
f(x)=x^2-4,
\]
state the domain and codomain, and determine the range.

\item Let
\[
f(x)=2x^2-3x+1.
\]
Compute \(f(0)\), \(f(-2)\), \(f(a)\), and \(f(x+h)\).

\item If
\[
g(x)=\frac{1}{x-3},
\]
determine the natural real domain of \(g\) and compute \(g(1)\).

\item The graph of a function contains the points
\[
(-2,3),\quad (0,-1),\quad (4,5).
\]
State the corresponding function values.

\item For
\[
f(x)=x^2-5x+6,
\]
find the \(y\)-intercept and all \(x\)-intercepts.

\item Determine the natural real domain of each function:
\[
f(x)=\sqrt{x+2},
\qquad
g(x)=\frac{x+1}{x^2-9},
\qquad
h(x)=\ln(4-x).
\]

\item Find the range of each function:
\[
f(x)=x^2+2,
\qquad
g(x)=-3x+1,
\qquad
h(x)=\sqrt{x}.
\]

\item Classify each function as constant, affine, quadratic, polynomial, rational, exponential, logarithmic, or trigonometric:
\[
3,\quad 2x-5,\quad x^2+1,\quad \frac{x+1}{x-2},\quad 4^x,\quad \ln x,\quad \sin x.
\]

\item Determine which of the functions
\[
f(x)=3x,\qquad g(x)=3x+2,\qquad h(x)=-x
\]
are linear in the sense \(f(x+y)=f(x)+f(y)\). Explain why an affine function need not be linear.

\item For
\[
f(x)=2^x
\quad\text{and}\quad
g(x)=\log_2 x,
\]
compute \(f(-2)\), \(f(3)\), \(g(1)\), \(g(8)\), and explain the relationship between the two functions.

\item Let
\[
a_n=3n-1,\qquad n\ge1.
\]
Write the first five terms and compute \(a_{20}\).

\item A sequence is defined by
\[
a_1=4,\qquad a_{n+1}=a_n-3.
\]
Write the first six terms.

\item Find an explicit formula for the arithmetic sequence
\[
5,\ 9,\ 13,\ 17,\ldots
\]

\item Find an explicit formula for the geometric sequence
\[
3,\ 6,\ 12,\ 24,\ldots
\]

\item Define recursively the sequence
\[
2,\ 6,\ 18,\ 54,\ldots
\]
and state the starting index.

\item For
\[
a_n=\frac{1}{n},
\]
list the first five graph points \((n,a_n)\). Explain why the graph should normally be drawn as separated points.

\item Compare the domains of
\[
a_n=n^2,\qquad n\in\mathbb{N},
\]
and
\[
f(x)=x^2,\qquad x\in\mathbb{R}.
\]
Which expressions \(a_{3/2}\) and \(f(3/2)\) are meaningful?

\item A graph rises on \((-3,1)\), falls on \((1,4)\), crosses the \(x\)-axis at \(-2\) and \(3\), and has a local maximum at \((1,5)\). Translate this information into statements about the function.

\item A taxi fare consists of a fixed fee of \(6\) units and \(2.5\) units for every kilometre travelled. Define a function \(C(d)\), state a meaningful domain, compute \(C(8)\), and explain whether the function is linear or affine.
\end{enumerate}

\section{Problem Set 9: Limits and Continuity}

\begin{enumerate}
\item Explain in words the difference between
\[
f(a)
\quad\text{and}\quad
\lim_{x\to a}f(x).
\]

\item Compute
\[
\lim_{n\to\infty}\frac{1}{n},
\qquad
\lim_{n\to\infty}\frac{5n+2}{2n-1}.
\]

\item Compute
\[
\lim_{n\to\infty}\frac{3n^2-n+1}{2-n^2}.
\]

\item Determine whether each sequence converges:
\[
a_n=(-1)^n,
\qquad
b_n=\frac{(-1)^n}{n},
\qquad
c_n=2+\frac{3}{n}.
\]

\item Compute
\[
\lim_{n\to\infty}\left(\sqrt{n^2+4n}-n\right)
\]
by rationalizing.

\item Compute by direct substitution:
\[
\lim_{x\to2}(3x^2-x+4).
\]

\item Compute
\[
\lim_{x\to3}\frac{x^2-9}{x-3}.
\]

\item Compute
\[
\lim_{x\to9}\frac{\sqrt{x}-3}{x-9}.
\]

\item Let
\[
f(x)=
\begin{cases}
2x+1,&x<2,\\
x^2-1,&x\ge2.
\end{cases}
\]
Find the left-hand limit, right-hand limit, two-sided limit, and value \(f(2)\).

\item Compute
\[
\lim_{x\to0^-}\frac{|x|}{x},
\qquad
\lim_{x\to0^+}\frac{|x|}{x},
\]
and decide whether the two-sided limit exists.

\item Compute
\[
\lim_{x\to\infty}\frac{4x^2-3x+1}{2x^2+5}.
\]

\item Find all horizontal asymptotes of
\[
f(x)=\frac{3x-1}{x+4}.
\]

\item Determine the one-sided infinite limits of
\[
f(x)=\frac{1}{x-2}
\]
as \(x\to2^-\) and \(x\to2^+\). State the vertical asymptote.

\item Check whether
\[
f(x)=\frac{x^2-4}{x-2},\qquad x\ne2,
\]
is continuous at \(x=2\). If not, give a value of \(f(2)\) that makes it continuous.

\item Find \(k\) so that
\[
f(x)=
\begin{cases}
kx+1,&x<3,\\
x^2-2,&x\ge3
\end{cases}
\]
is continuous at \(x=3\).

\item Classify the discontinuity at the indicated point:
\begin{enumerate}
\item \(\dfrac{x^2-1}{x-1}\) at \(x=1\);
\item \(\dfrac{1}{x+2}\) at \(x=-2\);
\item \(\dfrac{|x|}{x}\) at \(x=0\);
\item \(\sin(1/x)\) at \(x=0\).
\end{enumerate}

\item Compute
\[
\lim_{x\to0}\frac{\sin 5x}{2x}.
\]

\item Compute
\[
\lim_{x\to0}\frac{\tan 3x}{\sin 2x}.
\]

\item Compute
\[
\lim_{x\to0}(1+4x)^{1/x}.
\]

\item For each limit, identify the most appropriate first method and then compute:
\[
\lim_{x\to1}\frac{x^3-1}{x-1},
\qquad
\lim_{x\to\infty}\frac{2x+5}{x^2+1},
\qquad
\lim_{x\to0}\frac{\sin x}{x},
\qquad
\lim_{x\to4}\frac{x-4}{\sqrt{x}-2}.
\]
\end{enumerate}

\section{Problem Set 10: Derivatives and Function Analysis}

\begin{enumerate}
\item For
\[
f(x)=x^2+1,
\]
form and simplify the difference quotient
\[
\frac{f(a+h)-f(a)}{h}.
\]

\item Use the limit definition to find the derivative of
\[
f(x)=3x+2.
\]

\item Find the tangent line to
\[
f(x)=x^2-1
\]
at \(x=2\).

\item Differentiate, stating the relevant domain:
\[
f(x)=x^7,\qquad
g(x)=x^{-3},\qquad
h(x)=\sqrt{x}.
\]

\item Differentiate:
\[
f(x)=4e^x-3\ln x+2\sin x-\cos x.
\]

\item Differentiate using the product rule:
\[
f(x)=x^3e^x.
\]

\item Differentiate using the quotient rule:
\[
f(x)=\frac{x^2+1}{x+2}.
\]

\item Differentiate using the chain rule:
\[
f(x)=(2x-3)^6.
\]

\item Differentiate:
\[
f(x)=e^{\sin(x^2)}.
\]
Identify all layers of composition.

\item Find all critical points of
\[
f(x)=x^3-6x^2+9x.
\]

\item Study the monotonicity of
\[
f(x)=x^2-4x+1
\]
using a sign chart for \(f'\).

\item Find and classify the local extrema of
\[
f(x)=x^3-3x.
\]

\item Find the absolute maximum and minimum of
\[
f(x)=x^2-2x
\]
on \([-1,3]\).

\item Determine the intervals of convexity and concavity of
\[
f(x)=x^3-6x.
\]

\item Find all inflection points of
\[
f(x)=x^4-4x^3.
\]
Remember that \(f''(a)=0\) gives only a candidate.

\item Use the second derivative test to classify the critical points of
\[
f(x)=x^4-4x^2.
\]
For any inconclusive point, use another method.

\item Find all vertical and horizontal asymptotes of
\[
f(x)=\frac{2x+1}{x-3}.
\]

\item Perform a complete analysis of
\[
f(x)=x^3-3x^2.
\]
Include domain, zeros, behavior at infinity, monotonicity, extrema, convexity, and inflection points.

\item Perform a complete analysis of
\[
f(x)=\frac{x}{x-1}.
\]
Include domain, asymptotes, monotonicity, and a coherent description of the graph.

\item A quantity is modelled by
\[
Q(t)=t^3-6t^2+9t+4,\qquad t\ge0.
\]
Find \(Q'(t)\), determine when the quantity is increasing or decreasing, identify local extrema, and explain what the sign of \(Q'(t)\) means in a modelling context.
\end{enumerate}

\section{Problem Set 11: Integrals}

\begin{enumerate}
\item Find all antiderivatives of
\[
f(x)=6x^2-4.
\]

\item Compute, on appropriate intervals,
\[
\int x^5\,dx,
\qquad
\int x^{-4}\,dx,
\qquad
\int \sqrt{x}\,dx.
\]

\item Compute
\[
\int \left(3e^x-\frac{2}{x}+4\cos x\right)\,dx.
\]

\item Rewrite and integrate:
\[
\int \frac{\sqrt{x}}{x^3}\,dx.
\]

\item Use substitution to compute
\[
\int 2x(1+x^2)^5\,dx.
\]

\item Use substitution to compute
\[
\int \frac{e^x}{1+e^x}\,dx.
\]

\item Use integration by parts to compute
\[
\int xe^x\,dx.
\]

\item Compute
\[
\int x^2\cos x\,dx
\]
using integration by parts more than once.

\item Compute
\[
\int \frac{2x+3}{x^2+3x+5}\,dx.
\]

\item Compute
\[
\int \frac{x}{x^2-2x+5}\,dx
\]
by splitting the numerator into a derivative part and a remainder.

\item Compute
\[
\int \frac{x^2+2}{x-1}\,dx
\]
by polynomial division.

\item Compute
\[
\int \sin^4 x\cos x\,dx.
\]

\item Compute
\[
\int \cos^2 x\,dx
\]
using a trigonometric identity.

\item Evaluate
\[
\int_0^2 (3x+1)\,dx.
\]

\item Find the signed area represented by
\[
\int_{-2}^{2}x\,dx
\]
and compare it with the ordinary geometric area between \(y=x\) and the \(x\)-axis on \([-2,2]\).

\item Use properties of definite integrals to simplify:
\[
\int_0^5 f(x)\,dx-\int_2^5 f(x)\,dx,
\]
and
\[
\int_3^1 g(x)\,dx.
\]

\item Use the fundamental theorem of calculus to evaluate
\[
\int_1^4 \frac{1}{x}\,dx.
\]

\item Let
\[
A(x)=\int_0^x (t^2+1)\,dt.
\]
Find \(A'(x)\) and compute \(A(2)\).

\item Choose an appropriate integration method for each integral and justify the choice before computing:
\[
\int x\sin(x^2)\,dx,
\qquad
\int x\ln x\,dx,
\qquad
\int \frac{3x^2}{x^3+1}\,dx.
\]

\item Compute and verify by differentiation:
\[
\int \left(
\frac{2x}{1+x^2}
+x e^{x^2}
+\sin x\cos x
\right)\,dx.
\]
\end{enumerate}

\section{Problem Set 12: Applications of Calculus}

\begin{enumerate}
\item For each question, state whether the main tool is a derivative, a definite integral, optimisation, or a differential model:
\begin{enumerate}
\item find instantaneous velocity;
\item find total water discharged from a known flow rate;
\item choose dimensions giving the largest area;
\item model a population whose growth rate is proportional to its size.
\end{enumerate}

\item A rectangle has perimeter \(40\). Express its area as a function of one side and find the dimensions that maximise the area.

\item A rectangular enclosure is built beside a straight wall, so only three sides require fencing. If \(60\) metres of fencing are available, find the dimensions that maximise the area.

\item Squares of side \(x\) are cut from the corners of a \(20\text{ cm}\times30\text{ cm}\) sheet to form an open box. Build the volume function, determine the meaningful domain, and find the value of \(x\) that maximises the volume.

\item Find the area enclosed by
\[
y=4-x^2
\]
and the \(x\)-axis.

\item The curve
\[
y=\sqrt{x},\qquad 0\le x\le9,
\]
is rotated about the \(x\)-axis. Find the resulting volume.

\item Set up, but do not necessarily evaluate, the arc-length integral for
\[
y=x^2,\qquad 0\le x\le2.
\]

\item A particle has position
\[
s(t)=t^3-6t^2+9t,\qquad t\ge0.
\]
Find its velocity and acceleration.

\item For the particle in the previous problem, determine when it is at rest and on which intervals it is speeding up or slowing down.

\item A particle has acceleration
\[
a(t)=6t-4,
\]
initial velocity \(v(0)=3\), and initial position \(s(0)=2\). Find \(v(t)\) and \(s(t)\).

\item A spring obeys Hooke's law \(F(x)=40x\) newtons. Find the work required to stretch it from \(x=0.1\) m to \(x=0.25\) m.

\item Water flows into a tank at the non-negative rate
\[
r(t)=12-2t
\]
litres per minute for \(0\le t\le6\). How much water enters during the first four minutes?

\item The temperature over a six-hour interval is modelled by
\[
T(t)=18+2t-\frac{t^2}{3},\qquad 0\le t\le6.
\]
Find its average value on the interval.

\item A population satisfies
\[
N'(t)=0.04N(t),
\qquad
N(0)=500.
\]
Find \(N(t)\) and the doubling time.

\item A radioactive substance satisfies
\[
M'(t)=-0.08M(t),
\qquad
M(0)=120.
\]
Find \(M(t)\) and its half-life.

\item A quantity \(Q(t)\) has rate of change
\[
Q'(t)=3t^2-4t+1
\]
and \(Q(0)=7\). Find \(Q(t)\) and explain the role of the initial condition.

\item A closed cylindrical can must have fixed volume \(V\). Derive the relation between radius and height that minimises its surface area.

\item A factory is \(400\) m downstream on the opposite side of a river \(80\) m wide. Underwater cable costs three times as much per metre as land cable. Let \(x\) be the downstream distance from the point directly opposite the station to the landing point. Build the cost function and determine the optimal \(x\).

\item A car's velocity is
\[
v(t)=t^2-6t+8,\qquad 0\le t\le6.
\]
Find the signed displacement and the total distance travelled on the interval.

\item A company models daily profit by
\[
P(x)=-2x^3+30x^2-96x-50,
\qquad 0\le x\le10,
\]
where \(x\) is production in hundreds of units. Determine the production levels at which profit is locally maximal or minimal, compare all relevant endpoint and critical values, and interpret the economically best choice within the model.
\end{enumerate}
