\begin{frame}{Repeated products of a diagonal matrix}
  \[
  D=\operatorname{diag}(3,-1,2)
  \]
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Second product}
        \[
        D^2=\operatorname{diag}(9,1,4)
        \]
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Third product}
        \[
        D^3=\operatorname{diag}(27,-1,8)
        \]
      \end{exampleblock}
    \end{column}
  \end{columns}
  \vspace{0.45cm}
  \takeaway{Each diagonal position evolves separately.}
\end{frame}
