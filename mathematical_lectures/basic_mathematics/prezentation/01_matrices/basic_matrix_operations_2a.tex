\begin{frame}{Worked example: subtraction and scaling}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{$A-B$}
        \[
        \begin{pmatrix}2&-1\\0&3\end{pmatrix}
        -
        \begin{pmatrix}5&4\\-2&1\end{pmatrix}
        =
        \begin{pmatrix}-3&-5\\2&2\end{pmatrix}
        \]
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{$-3A$}
        \[
        -3\begin{pmatrix}2&-1\\0&3\end{pmatrix}
        =
        \begin{pmatrix}-6&3\\0&-9\end{pmatrix}
        \]
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Linear combinations: build a new matrix}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.42\textwidth}
      \[
      \alpha A+\beta B
      \]
      \begin{block}{Interpretation}
        scale two matrices, then combine matching positions
      \end{block}
    \end{column}
    \begin{column}{0.54\textwidth}
      \[
      2\begin{pmatrix}1&0\\-1&2\end{pmatrix}
      -
      \begin{pmatrix}3&1\\2&0\end{pmatrix}
      =
      \begin{pmatrix}-1&-1\\-4&4\end{pmatrix}
      \]
    \end{column}
  \end{columns}
\end{frame}
