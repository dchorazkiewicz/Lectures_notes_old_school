\begin{frame}{Correct sizes for $I$}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Left side}
        \[
        A\in\R^{m\times n},\qquad I_mA=A
        \]
        $I_m$ has size $m\times m$.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Right side}
        \[
        AI_n=A
        \]
        $I_n$ has size $n\times n$.
      \end{exampleblock}
    \end{column}
  \end{columns}
  \vspace{0.55cm}
  \takeaway{The side of multiplication determines the required size.}
\end{frame}
